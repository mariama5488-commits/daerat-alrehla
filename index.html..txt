    <script>
        const referenceGregorianDate = new Date(2026, 4, 5); 

        const taskConfig = [
            { id: 'quran', cat: 'ديني', title: 'وِرد القرآن والأذكار', freq: 'daily' },
            { id: 'esp', cat: 'تعلم', title: 'لسان الإسبان', freq: 'daily' },
            { id: 'women', cat: 'ثقافة', title: 'طبائع النساء ومطالعة', freq: ['Saturday', 'Monday', 'Wednesday'] },
            { id: 'ai', cat: 'مستقبل', title: 'مختبر الذكاء الاصطناعي', freq: ['Sunday', 'Thursday'] },
            { id: 'icu', cat: 'طب', title: 'حالات الـ ICU واللوج بوك', freq: ['Tuesday'] },
            { id: 'gym', cat: 'صحة', title: 'إعداد البدن', freq: ['Tuesday', 'Friday'] }
        ];

        const dayRules = { "Saturday": 2, "Sunday": 5, "Monday": 0, "Tuesday": 5, "Wednesday": 0, "Thursday": 5, "Friday": 5 };

        let db = JSON.parse(localStorage.getItem('daerat_alrehla_db_v5')) || { done: {}, refs: {}, golden: {}, habits: {} };
        
        let selectedDate = new Date(2026, 4, 5); 
        let viewDate = new Date(2026, 4, 5); 
        
        let timer = null;
        let timeLeft = 1500;
        let activeTask = null;

        function save() { localStorage.setItem('daerat_alrehla_db_v5', JSON.stringify(db)); updateStats(); }
        function getDateKey(d) { return `${d.getFullYear()}-${d.getMonth()+1}-${d.getDate()}`; }

        function calculateHijri(targetDate) {
            try {
                const formatter = new Intl.DateTimeFormat('ar-SA-u-ca-islamic-umalqura', {
                    day: 'numeric', month: 'long', year: 'numeric'
                });
                return formatter.format(targetDate);
            } catch(e) {
                const diffTime = targetDate.getTime() - referenceGregorianDate.getTime();
                const diffDays = Math.round(diffTime / (1000 * 60 * 60 * 24));
                let hDay = 17 + diffDays;
                if (hDay > 30) { hDay -= 30; return `${hDay} ذو الحجة 1447 هـ`; }
                if (hDay < 1) { hDay += 30; return `${hDay} شوال 1447 هـ`; }
                return `${hDay} ذو القعدة 1447 هـ`;
            }
        }

        function changeMonth(dir) {
            viewDate.setMonth(viewDate.getMonth() + dir);
            renderCalendar();
        }

        function renderCalendar() {
            const rail = document.getElementById('calendar-rail');
            const monthDisplay = document.getElementById('current-month-display');
            const hijriMonthDisplay = document.getElementById('hijri-month-display');
            if (!rail) return;

            rail.innerHTML = '';
            monthDisplay.innerText = viewDate.toLocaleDateString('ar-EG', { month: 'long', year: 'numeric' });
            
            const middleDay = new Date(viewDate.getFullYear(), viewDate.getMonth(), 15);
            const hFull = calculateHijri(middleDay);
            const hParts = hFull.split(' ');
            hijriMonthDisplay.innerText = hParts.length > 2 ? hParts.slice(1).join(' ') : "ذو القعدة 1447 هـ";

            const year = viewDate.getFullYear();
            const month = viewDate.getMonth();
            const daysInMonth = new Date(year, month + 1, 0).getDate();

            for (let i = 1; i <= daysInMonth; i++) {
                const d = new Date(year, month, i);
                const key = getDateKey(d);
                const isActive = key === getDateKey(selectedDate);
                
                const hijriFull = calculateHijri(d);
                const hijriMatch = hijriFull.match(/\d+/);
                const hijriDayNum = hijriMatch ? hijriMatch[0] : i;

                const el = document.createElement('div');
                el.className = `day-arch ${isActive ? 'active-day' : ''}`;
                el.onclick = () => { 
                    selectedDate = new Date(d); 
                    renderCalendar(); 
                    renderDay(); 
                };
                el.innerHTML = `
                    <span class="text-[10px] font-bold mb-1 opacity-50">${d.toLocaleDateString('ar-EG', {weekday: 'short'})}</span>
                    <span class="text-2xl font-bold font-tradition leading-none">${i}</span>
                    <span class="text-[10px] mt-2 font-amiri text-[var(--royal-gold)] font-bold">${hijriDayNum}</span>
                `;
                rail.appendChild(el);
                if (isActive) el.scrollIntoView({ behavior: 'smooth', inline: 'center', block: 'nearest' });
            }
        }

        function renderDay() {
            const key = getDateKey(selectedDate);
            const dayEn = selectedDate.toLocaleDateString('en-US', {weekday:'long'});
            
            document.getElementById('day-name').innerText = selectedDate.toLocaleDateString('ar-EG', {weekday:'long'});
            document.getElementById('full-date').innerText = selectedDate.toLocaleDateString('ar-EG', {day:'numeric', month:'long', year:'numeric'});
            document.getElementById('hijri-date').innerText = calculateHijri(selectedDate);

            const r = db.refs[key] || {};
            document.getElementById('ref-emotions').value = r.emotions || '';
            document.getElementById('ref-body').value = r.body || '';
            document.getElementById('ref-gratitude').value = r.gratitude || '';
            document.getElementById('ref-idea').value = r.idea || '';
            
            document.getElementById('habit-check').innerText = db.habits[key] ? 'تم التثبيت ✨' : 'ثبات السعي';

            const grid = document.getElementById('tasks-grid');
            grid.innerHTML = '';
            
            if(dayRules[dayEn] === 0) {
                grid.innerHTML = `<div class="col-span-full py-20 text-center animate-fade"><p class="font-amiri text-2xl text-slate-400 italic opacity-40">🏥 يوم الميدان.. كان الله في عونكِ.</p></div>`;
                return;
            }

            taskConfig.forEach(t => {
                if(t.freq === 'daily' || t.freq.includes(dayEn)) {
                    const taskStatus = db.done[key + t.id];
                    const isDone = !!taskStatus;
                    const noteSnippet = taskStatus?.note ? taskStatus.note : "";
                    const isGolden = db.golden[key] === t.id;
                    
                    const card = document.createElement('div');
                    card.className = `glass-card p-6 cursor-pointer hover:shadow-xl transition-all relative overflow-hidden ${isDone ? 'opacity-60 bg-green-50/20' : ''} ${isGolden ? 'border-2 border-[var(--royal-gold)] ring-4 ring-[var(--royal-gold)]/5' : ''}`;
                    card.onclick = () => openTask(t);
                    card.innerHTML = `
                        <div class="flex justify-between mb-2">
                            <span class="text-[10px] font-cairo font-bold text-[var(--henna-red)] opacity-60">${t.cat}</span>
                            ${isGolden ? '<span>⭐</span>' : ''}
                        </div>
                        <h4 class="text-xl font-tradition text-[var(--deep-indigo)] mb-2">${t.title}</h4>
                        ${noteSnippet ? `<div class="bg-[var(--royal-gold)]/5 p-2 rounded-lg border-r-2 border-[var(--royal-gold)] mt-2"><p class="text-[11px] text-slate-500 font-amiri leading-tight line-clamp-2 italic"><span class="font-bold">توقفتِ عند:</span> ${noteSnippet}</p></div>` : ''}
                        <div class="flex justify-end mt-4"><div class="w-7 h-7 rounded-full border-2 border-slate-100 flex items-center justify-center transition-all ${isDone ? 'bg-green-500 border-none text-white scale-110' : ''}">${isDone ? '✓' : ''}</div></div>
                    `;
                    grid.appendChild(card);
                }
            });
        }

        function toggleSidebar() { document.getElementById('sidebar').classList.toggle('closed'); }
        function openTask(t) { activeTask = t; const key = getDateKey(selectedDate); document.getElementById('task-title-modal').innerText = t.title; document.getElementById('task-note').value = db.done[key + t.id]?.note || ''; document.getElementById('task-modal').classList.remove('hidden'); }
        function closeTask() { document.getElementById('task-modal').classList.add('hidden'); }
        function toggleGoldenTask() { const key = getDateKey(selectedDate); db.golden[key] = (db.golden[key] === activeTask.id) ? null : activeTask.id; save(); renderDay(); closeTask(); }
        function finishTask() { const key = getDateKey(selectedDate); db.done[key + activeTask.id] = { note: document.getElementById('task-note').value }; save(); closeTask(); renderDay(); }
        function saveRef(field, val) { const key = getDateKey(selectedDate); if(!db.refs[key]) db.refs[key] = {}; db.refs[key][field] = val; save(); }
        function markHabit() { const key = getDateKey(selectedDate); db.habits[key] = !db.habits[key]; save(); renderDay(); }

        function toggleTimer() {
            const btn = document.getElementById('timer-btn');
            const ring = document.getElementById('astrolabe-progress');
            if(timer) { clearInterval(timer); timer = null; btn.innerText = 'استئناف'; document.body.classList.remove('focus-mode'); } 
            else {
                document.body.classList.add('focus-mode'); btn.innerText = 'إيقاف';
                timer = setInterval(() => {
                    if(timeLeft > 0) { timeLeft--; document.getElementById('timer-text').innerText = `${Math.floor(timeLeft/60)}:${(timeLeft%60).toString().padStart(2, '0')}`; ring.style.strokeDashoffset = 226 - (timeLeft/1500)*226; } 
                    else { clearInterval(timer); alert('انتهت الخلوة.'); document.body.classList.remove('focus-mode'); btn.innerText = 'بدء'; timeLeft = 1500; }
                }, 1000);
            }
        }

        function updateStats() {
            const container = document.getElementById('stats-bars');
            if (!container) return;
            const stats = [{ label: 'العلوم الطبية', color: 'var(--henna-red)', val: 65 }, { label: 'الذكاء الاصطناعي', color: 'var(--deep-indigo)', val: 40 }, { label: 'اللغة الإسبانية', color: 'var(--royal-gold)', val: 80 }];
            container.innerHTML = '';
            stats.forEach(s => { container.innerHTML += `<div class="space-y-2"><div class="flex justify-between text-[11px] font-bold"><span>${s.label}</span><span>${s.val}%</span></div><div class="w-full h-1 bg-slate-100 rounded-full overflow-hidden"><div class="h-full" style="width: ${s.val}%; background-color: ${s.color}"></div></div></div>`; });
        }

        window.onload = () => { renderCalendar(); renderDay(); updateStats(); };
    </script>
